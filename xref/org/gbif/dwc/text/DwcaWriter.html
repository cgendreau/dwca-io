47   * The meta.xml descriptor is generated automatically and an optional EML metadata document can be added.

119   private void addRowType(Term rowType) throws IOException {
120     terms.put(rowType, new ArrayList<Term>());
121 
122     String dfn = dataFileName(rowType);
123     dataFileNames.put(rowType, dfn);
124     File df = new File(dir, dfn);
125     FileUtils.forceMkdir(df.getParentFile());
126     OutputStream out = new FileOutputStream(df);
127     TabWriter wr = new TabWriter(out);
128     writers.put(rowType, wr);
129   }

155   private void writeRow(Map<Term, String> rowMap, Term rowType) throws IOException {
156     TabWriter writer = writers.get(rowType);
157     List<Term> columns = terms.get(rowType);
158     if (useHeaders && !headersOut.contains(rowType)){
159       // write header row
160       writeHeader(writer, rowType, columns);
161     }
162 
163     // make sure coreId is not null for extensions
164     if (coreRowType != rowType && coreId == null){
165       log.warn("Adding an {} extension record to a core without an Id! Skip this record", rowType);
166 
167     } else {
168       String[] row = new String[columns.size() + 1];
169       row[0] = coreId;
170       for (Map.Entry<Term, String> conceptTermStringEntry : rowMap.entrySet()) {
171         int column = 1 + columns.indexOf(conceptTermStringEntry.getKey());
172         row[column] = conceptTermStringEntry.getValue();
173       }
174       writer.write(row);
175     }
176   }

178   private void writeHeader(TabWriter writer, Term rowType, List<Term> columns) throws IOException {
179     int idx = 0;
180     String[] row = new String[columns.size() + 1];
181     Term idTerm;
182     if (DwcTerm.Taxon == coreRowType){
183       idTerm = DwcTerm.taxonID;
184     } else if (DwcTerm.Occurrence == coreRowType){
185       idTerm = DwcTerm.occurrenceID;
186     } else if (DwcTerm.Identification == coreRowType){
187       idTerm = DwcTerm.identificationID;
188     } else if (DwcTerm.Event == coreRowType){
189       idTerm = DwcTerm.eventID;
190     } else {
191       // default to generic dc identifier for id column
192       idTerm = DcTerm.identifier;
193     }
194     row[idx] = idTerm.simpleName();
195 
196     for (Term term : columns) {
197       idx ++;
198       row[idx] = term.simpleName();
199     }
200     writer.write(row);
201 
202     headersOut.add(rowType);
203   }

206   /**
207    * Add a single value for the current core record.
208    * Calling this method requires that #newRecord() has been called at least once,
209    * otherwise an IllegalStateException is thrown.
210    * @param term
211    * @param value
212    */
213   public void addCoreColumn(Term term, String value) {
214     // ensure we do not overwrite the coreIdTerm if one is defined
215     if (coreIdTerm != null && coreIdTerm.equals(term)) {
216       throw new IllegalStateException("You cannot add a term that was specified as coreId term");
217     }
218     
219     // ensure no default value is already associated with this term 
220     Map<Term,String> coreDefaultValues = defaultValues.get(coreRowType);
221     if(coreDefaultValues !=null && coreDefaultValues.containsKey(term)){
222       throw new IllegalStateException("A default value for term "+ term + " is already defined");
223     }
224     
225     List<Term> coreTerms = terms.get(coreRowType);
226     if (!coreTerms.contains(term)) {
227       if (useHeaders && recordNum>1){
228         throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
229       }
230       coreTerms.add(term);
231     }
232     try {
233       coreRow.put(term, value);
234     } catch (NullPointerException e) {
235       // no core record has been started yet
236       throw new IllegalStateException("No core record has been created yet. Call newRecord() at least once");
237     }
238   }

250   /**
251    * Add a default value to a term of the provided rowType.
252    * 
253    * @param rowType
254    * @param term
255    * @param defaultValue
256    */
257   public void addDefaultValue(Term rowType, Term term, String defaultValue){
258     List<Term> knownTerms = terms.get(rowType);
259     
260     // avoid overwriting a term that was already defined
261     if(knownTerms !=null && knownTerms.contains(term)){
262       throw new IllegalStateException("The term "+ term + " is already defined for rowType " + rowType );
263     }
264     
265     if(!defaultValues.containsKey(rowType)){
266       defaultValues.put(rowType, new HashMap<Term, String>());
267     }
268     Map<Term,String> currentDefaultValues= defaultValues.get(rowType);
269     if(currentDefaultValues.containsKey(term)){
270       throw new IllegalStateException("The default value of term "+ term + " is already defined");
271     }
272     currentDefaultValues.put(term, defaultValue);
273   }

282   /**
283    * Add an extension record associated with the current core record.
284    * 
285    * @param rowType
286    * @param row
287    * @throws IOException
288    */
289   public void addExtensionRecord(Term rowType, Map<Term, String> row) throws IOException {
290     // make sure we know the extension rowtype
291     if (!terms.containsKey(rowType)) {
292       addRowType(rowType);
293     }
294     
295     // make sure no default value is already associated with one of the provided terms
296     Map<Term,String> extDefaultValues = defaultValues.get(rowType);
297     if(extDefaultValues !=null){
298       Set<Term> diff = Sets.intersection(extDefaultValues.keySet(), row.keySet());
299       if(!diff.isEmpty()){
300         throw new IllegalStateException("A default value is already defined for term(s) "+
301           diff.toString() + " in rowType " + rowType);
302       }
303     }
304     
305     // make sure we know all terms
306     List<Term> knownTerms = terms.get(rowType);
307     final boolean isFirst = knownTerms.isEmpty();
308     for (Term term : row.keySet()) {
309       if (!knownTerms.contains(term)) {
310         if (useHeaders && !isFirst){
311           throw new IllegalStateException("You cannot add new terms after the first row when headers are enabled");
312         }
313         knownTerms.add(term);
314       }
315     }
316 
317     // write extension record
318     writeRow(row, rowType);
319   }

351   private void addEml() throws IOException {
352     if (eml != null) {
353       File emlFile = new File(dir, "eml.xml");
354       try {
355         EmlWriter.writeEmlFile(emlFile, eml);
356       } catch (TemplateException e) {
357         throw new IOException("EML template exception: " + e.getMessage());
358       }

362   private void addMeta() throws IOException {
363     File metaFile = new File(dir, "meta.xml");
364 
365     Archive arch = new Archive();
366     if (eml != null) {
367       arch.setMetadataLocation("eml.xml");
368     }
369     arch.setCore(buildArchiveFile(arch, coreRowType, coreIdTerm));
370     for (Term rowType : this.terms.keySet()) {
371       if (!coreRowType.equals(rowType)) {
372         arch.addExtension(buildArchiveFile(arch, rowType, null));
373       }
374     }
375     try {
376       MetaDescriptorWriter.writeMetaFile(metaFile, arch);
377     } catch (TemplateException e) {
378       throw new IOException("Meta.xml template exception: " + e.getMessage());
379     }
380   }

382   private ArchiveFile buildArchiveFile(Archive archive, Term rowType, Term idTerm) {
383     ArchiveFile af = ArchiveFile.buildTabFile();
384     af.setArchive(archive);
385     af.addLocation(dataFileNames.get(rowType));
386 
387     af.setEncoding("utf-8");
388     af.setIgnoreHeaderLines(useHeaders ? 1 : 0);
389     af.setRowType(rowType.qualifiedName());
390 
391     ArchiveField id = new ArchiveField();
392     id.setIndex(0);
393     af.setId(id);
394     // id an idTerm is provided, always use the index 0
395     if (idTerm != null) {
396       ArchiveField field = new ArchiveField();
397       field.setIndex(0);
398       field.setTerm(idTerm);
399       af.addField(field);
400     }
401     
402     int idx = 0;
403     for (Term c : terms.get(rowType)) {
404       idx++;
405       ArchiveField field = new ArchiveField();
406       field.setIndex(idx);
407       field.setTerm(c);
408       af.addField(field);
409     }
410     
411     // check if default values are provided for this rowType
412     Map<Term,String> termDefaultValueMap = defaultValues.get(rowType);
413     if(termDefaultValueMap != null){
414       ArchiveField field = null;
415       for (Term t : termDefaultValueMap.keySet()) {
416         field = new ArchiveField();
417         field.setTerm(t);
418         field.setDefaultValue(termDefaultValueMap.get(t));
419         af.addField(field);
420       }
421     }
422 
423     return af;
424   }
