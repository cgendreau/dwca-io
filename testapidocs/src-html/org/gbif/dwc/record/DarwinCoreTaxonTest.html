017  @Test
018  public void testGetFullScientificName() {
019    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
020    dwc.setScientificNameAuthorship("A. Braun");
021    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
022    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
023
024    dwc = new DarwinCoreTaxon("Festuca x brinkmannii A. Braun");
025    dwc.setScientificNameAuthorship("A.Braun");
026    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
027    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
028
029    dwc = new DarwinCoreTaxon("Festuca x brinkmannii");
030    dwc.setScientificNameAuthorship("A. Braun");
031    assertEquals("Festuca x brinkmannii A. Braun", dwc.getFullScientificName());
032    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
033
034    dwc = new DarwinCoreTaxon("Festuca A. Braun brinkmannii subsp. brinkmannii");
035    dwc.setScientificNameAuthorship("A. Braun");
036    assertEquals("Festuca A. Braun brinkmannii subsp. brinkmannii", dwc.getFullScientificName());
037    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
038
039    dwc = new DarwinCoreTaxon("Festuca brinkmannii");
040    dwc.setScientificNameAuthorship("A. Braun");
041    assertEquals("Festuca brinkmannii A. Braun", dwc.getFullScientificName());
042    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
043
044    dwc = new DarwinCoreTaxon(" ");
045    dwc.setScientificNameAuthorship("Rafinesque, 1795");
046    dwc.setKingdom("Plantae");
047    assertNull(dwc.getFullScientificName());
048    assertEquals("Plantae", dwc.getLowestScientificName());
049
050    dwc = new DarwinCoreTaxon(" ");
051    dwc.setScientificNameAuthorship("L.");
052    dwc.setKingdom("Plantae");
053    dwc.setFamily("Asteraceae");
054    assertNull(dwc.getFullScientificName());
055    assertEquals("Asteraceae", dwc.getLowestScientificName());
056
057    dwc = new DarwinCoreTaxon();
058    dwc.setScientificNameAuthorship("L.");
059    dwc.setKingdom("Plantae");
060    dwc.setGenus("Poa");
061    assertEquals("Poa", dwc.getFullScientificName());
062    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
063
064    dwc.setSpecificEpithet("annua");
065    assertEquals("Poa annua L.", dwc.getFullScientificName());
066    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
067
068    dwc.setScientificNameAuthorship("");
069    assertEquals("Poa annua", dwc.getFullScientificName());
070    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
071
072    dwc.setScientificNameAuthorship("L.");
073    dwc.setInfraspecificEpithet("alpina");
074    assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
075    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
076
077    dwc.setVerbatimTaxonRank("var.");
078    assertEquals("Poa annua alpina L.", dwc.getFullScientificName());
079    assertEquals(dwc.getFullScientificName(), dwc.getLowestScientificName());
080
081    // test not identical authorships
082    dwc = new DarwinCoreTaxon();
083    dwc.setScientificName("Aagaardia protensa Saether, 2000");
084    dwc.setGenus("Aagaardia");
085    dwc.setSpecificEpithet("protensa");
086    dwc.setScientificNameAuthorship("Saether 2000");
087    assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
088
089    // test genus
090    dwc = new DarwinCoreTaxon();
091    dwc.setScientificName("Aagaardia protensa Saether, 2000");
092    dwc.setGenus("Aagaardia");
093    dwc.setSpecificEpithet("protensa");
094    dwc.setScientificNameAuthorship("Saether 2000");
095    assertEquals(dwc.getScientificName(), dwc.getFullScientificName());
096
097    // test authors with single letter that is part of the name
098    dwc = new DarwinCoreTaxon();
099    dwc.setScientificName("Trifolium repens");
100    dwc.setGenus("Trifolium");
101    dwc.setSpecificEpithet("repens");
102    dwc.setScientificNameAuthorship("L.");
103    assertEquals("Trifolium repens", dwc.getScientificName());
104    assertEquals("Trifolium repens L.", dwc.getFullScientificName());
105
106    dwc = new DarwinCoreTaxon();
107    dwc.setScientificName("Trifolium thalii");
108    dwc.setScientificNameAuthorship("Vill.");
109    assertEquals("Trifolium thalii Vill.", dwc.getFullScientificName());
110
111    dwc = new DarwinCoreTaxon();
112    dwc.setScientificName("Trifolium alpestre");
113    dwc.setScientificNameAuthorship("L.");
114    assertEquals("Trifolium alpestre L.", dwc.getFullScientificName());
115
116    // autonyms ???
117    dwc = new DarwinCoreTaxon();
118    dwc.setScientificName("Trifolium alpestre alpestre");
119    dwc.setScientificNameAuthorship("L.");
120    // this would be really correct for autonyms...
121    //assertEquals("Trifolium alpestre L. alpestre", dwc.getFullScientificName());
122    assertEquals("Trifolium alpestre alpestre L.", dwc.getFullScientificName());
123  }

125  @Test
126  public void testInterpretedRank() {
127    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
128    dwc.setKingdom("PLANTAE");
129    dwc.setPhylum("Magnoliophyta");
130    dwc.setFamily("Poaceae");
131    dwc.setGenus("Festuca");
132    dwc.setTaxonRank("species");
133    assertEquals("species", dwc.getTaxonRankInterpreted());
134
135    dwc.setSpecificEpithet("brinkmannii");
136    assertEquals("species", dwc.getTaxonRankInterpreted());
137
138    dwc.setInfraspecificEpithet("nonexistii");
139    assertEquals("species", dwc.getTaxonRankInterpreted());
140
141    dwc.setScientificName(null);
142    assertEquals("species", dwc.getTaxonRankInterpreted());
143
144    dwc.setTaxonRank(null);
145    assertEquals("infraspecies", dwc.getTaxonRankInterpreted());
146
147    dwc.setVerbatimTaxonRank("var.");
148    assertEquals("var.", dwc.getTaxonRankInterpreted());
149
150    dwc.setSpecificEpithet(null);
151    assertEquals("genus", dwc.getTaxonRankInterpreted());
152
153    dwc.setGenus(null);
154    assertEquals("family", dwc.getTaxonRankInterpreted());
155  }

157  @Test
158  public void testHigherClassification() {
159    DarwinCoreTaxon dwc = new DarwinCoreTaxon("Festuca brinkmannii A. Braun");
160    dwc.setKingdom("PLANTAE");
161    dwc.setPhylum("Magnoliophyta");
162    dwc.setFamily("Poaceae");
163    dwc.setGenus("Festuca");
164    assertEquals("Plantae | Magnoliophyta | Poaceae | Festuca", dwc.buildHigherClassification(null));
165    assertEquals("Plantae | Magnoliophyta | Poaceae", dwc.buildHigherClassification('f'));
166    assertEquals("Plantae | Magnoliophyta", dwc.buildHigherClassification('o'));
167  }

169  @Test
170  public void testProperties() {
171    Set<String> groups = ImmutableSet.of(DwcTerm.GROUP_TAXON, DwcTerm.GROUP_RECORD);
172    DarwinCoreTaxon dwc = new DarwinCoreTaxon();
173    for (DwcTerm t : DwcTerm.values()) {
174      if (groups.contains(t.getGroup()) && !t.isClass()){
175        String val = new Date().toString();
176        dwc.setProperty(t, val);
177        try {
178          dwc.getProperty(t);
179          assertEquals(val, dwc.getProperty(t));
180        } catch (Exception e) {
181          System.err.println(e.getMessage());
182        }
183      }
184    }
185  }
