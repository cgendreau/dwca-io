079  public String buildHigherClassification(Character lowestRank) {
080    StringBuilder buf = new StringBuilder();
081    if (kingdom != null) {
082      buf.append(StringUtils.capitalize(kingdom.toLowerCase()));
083      buf.append(CLASSIFICATION_DELIMITER);
084    }
085    if (lowestRank == null || lowestRank != 'k') {
086      if (phylum != null) {
087        buf.append(StringUtils.capitalize(phylum.toLowerCase()));
088        buf.append(CLASSIFICATION_DELIMITER);
089      }
090      if (lowestRank == null || lowestRank != 'p') {
091        if (classs != null) {
092          buf.append(StringUtils.capitalize(classs.toLowerCase()));
093          buf.append(CLASSIFICATION_DELIMITER);
094        }
095        if (lowestRank == null || lowestRank != 'c') {
096          if (order != null) {
097            buf.append(StringUtils.capitalize(order.toLowerCase()));
098            buf.append(CLASSIFICATION_DELIMITER);
099          }
100          if (lowestRank == null || lowestRank != 'o') {
101            if (family != null) {
102              buf.append(StringUtils.capitalize(family.toLowerCase()));
103              buf.append(CLASSIFICATION_DELIMITER);
104            }
105            if (lowestRank == null || lowestRank != 'f') {
106              if (genus != null) {
107                buf.append(StringUtils.capitalize(genus.toLowerCase()));
108                buf.append(CLASSIFICATION_DELIMITER);
109              }
110              if (lowestRank == null || lowestRank != 'g') {
111                if (subgenus != null) {
112                  buf.append(StringUtils.capitalize(subgenus.toLowerCase()));
113                  buf.append(CLASSIFICATION_DELIMITER);
114                }
115              }
116            }
117          }
118        }
119      }
120    }
121
122    if (buf.length() > CLASSIFICATION_DELIMITER.length()) {
123      buf.delete(buf.length() - CLASSIFICATION_DELIMITER.length(), buf.length());
124    }
125    return buf.toString();
126  }

128  public String getFullScientificName() {
129    if (scientificNameAuthorship != null && scientificName != null) {
130      String normedSciName = NORM_AUTHORS.matcher(scientificName.toLowerCase()).replaceAll(" ");
131      String normedAuthors = NORM_AUTHORS.matcher(scientificNameAuthorship.toLowerCase()).replaceAll(" ");
132      if (!normedSciName.contains(normedAuthors)) {
133        return scientificName + " " + scientificNameAuthorship;
134      }
135    }
136    if (scientificName == null) {
137      String sciname = null;
138      if (this.genus != null) {
139        if (specificEpithet != null) {
140          sciname = genus + " " + specificEpithet;
141          if (infraspecificEpithet != null) {
142            sciname += " " + infraspecificEpithet;
143          }
144          // potentially add authorship in this case
145          if (sciname != null && scientificNameAuthorship != null) {
146            sciname += " " + scientificNameAuthorship;
147          }
148        } else {
149          sciname = subgenus != null ? subgenus : genus;
150        }
151      }
152      return sciname;
153    }
154    return scientificName;
155  }
156
157  public String getLowestScientificName() {
158    String sciname = getFullScientificName();
159    if (sciname == null) {
160      if (family != null) {
161        sciname = family;
162      } else if (order != null) {
163        sciname = order;
164      } else if (classs != null) {
165        sciname = classs;
166      } else if (phylum != null) {
167        sciname = phylum;
168      } else if (kingdom != null) {
169        sciname = kingdom;
170      }
171    }
172    return sciname;
173  }

175  /**
176   * Gets a dwc property by its concept term.
177   * This method is only able to access official Darwin Core or Dublin Core terms, not any custom extensions.
178   *
179   * @param prop the concept term to lookup
180   *
181   * @return the terms value, null or IllegalArgumentException for unsupported terms
182   */
183  public String getProperty(Term prop) {
184    String getter = String.format("get%s", getPropertyName(prop));
185    try {
186      Method m = getClass().getMethod(getter);
187      Object val = m.invoke(this);
188      return val == null ? null : val.toString();
189
190    } catch (NoSuchMethodException e) {
191      throw new IllegalArgumentException("non existing dwc property: " + prop);
192
193    } catch (Exception e) {
194      // should never happen
195      throw new RuntimeException(e);
196    }
197  }
198
199  private String getPropertyName(Term prop) {
200    String propName = StringUtils.capitalize(prop.simpleName());
201    if (propName.equalsIgnoreCase("Class")) {
202      propName = "Classs";
203    }
204    return propName;
205  }

207  public String getTaxonRankInterpreted() {
208    if (scientificName != null || taxonRank != null) {
209      return taxonRank;
210    }
211    if (genus != null && specificEpithet != null && infraspecificEpithet != null) {
212      if (taxonRank != null) {
213        return taxonRank;
214      }
215      if (verbatimTaxonRank != null) {
216        return verbatimTaxonRank;
217      }
218      return "infraspecies";
219    }
220    if (genus != null && specificEpithet != null) {
221      return "species";
222    }
223    if (subgenus != null) {
224      return "subgenus";
225    }
226    if (genus != null) {
227      return "genus";
228    }
229    if (family != null) {
230      return "family";
231    }
232    if (order != null) {
233      return "order";
234    }
235    if (classs != null) {
236      return "class";
237    }
238    if (phylum != null) {
239      return "phylum";
240    }
241    if (kingdom != null) {
242      return "kingdom";
243    }
244    return null;
245  }

247  /**
248   * The method tries to best assemble the most complete scientific name possible incl the name authorship.
249   * It first tries to use scientificName and scientificNameAuthorship if existing.
250   * Otherwise it uses the atomized name parts. Warning: this uses the genus property which might be wrong in case
251   * of synonym names, see https://code.google.com/p/darwincore/issues/detail?id=151
252   * Note also that the assembled name never includes a rank marker.
253   *
254   * @return the best guess of the full scientific name with authorship
255   */
256  protected String norm(String x) {
257    x = StringUtils.trimToNull(x);
258    if (x != null && (x.equalsIgnoreCase("\\N") || x.equalsIgnoreCase("NULL"))) {
259      x = null;
260    }
261    return x;
262  }

264  public boolean setProperty(Term prop, String value) {
265    if (prop == null) {
266      return false;
267    }
268    String setterName = null;
269    try {
270      setterName = String.format("set%s", getPropertyName(prop));
271      Method m = getClass().getMethod(setterName, String.class);
272      m.invoke(this, value);
273    } catch (Exception e) {
274      LOG.warn("Trying to access bad dwc property setter: " + setterName + " for property " + prop);
275      return false;
276    }
277    return true;
278  }
